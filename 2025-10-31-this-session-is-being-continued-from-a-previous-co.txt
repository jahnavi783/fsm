  Official Flutter Approach for Custom Theme Data:

  Flutter provides exactly two official patterns for passing data globally through the widget tree:

  1. ThemeExtension<T> (Recommended for theme-related data)
    - Official API specifically designed for custom theme properties
    - Integrates with ThemeData.extensions
    - Supports theme transitions, hot-reload, and Material 3
    - This is what we implemented with SpacingTheme вң…
  2. InheritedWidget (Lower-level, for non-theme data)
    - More generic data propagation pattern
    - Requires manual implementation of dependOnInheritedWidgetOfExactType
    - Not integrated with Flutter's theme system
    - Not recommended for spacing/design tokens

  3. Why Not Use ThemeData Directly?

  The docs show that ThemeData only has built-in properties for standard Material components:

  There is NO built-in property for custom spacing tokens in ThemeData. That's why ThemeExtension was created - to allow apps to add custom theme properties while still integrating with Flutter's theme
  system.

    static const dark = SpacingTheme(

    @override
    ThemeExtension<SpacingTheme> copyWith({
      double? space2,
      double? space4,
      double? buttonHeightMd,
      double? cardPadding,
      double? cardRadius,
    }) {
      return SpacingTheme(
        space2: space2 ?? this.space2,
        space4: space4 ?? this.space4,
        buttonHeightMd: buttonHeightMd ?? this.buttonHeightMd,

        cardRadius: cardRadius ?? this.cardRadius,

      ThemeExtension<SpacingTheme>? other,
      double t,

     /// Semantic text wrapper that automatically uses theme typography
     class FsmText extends StatelessWidget {
       const FsmText(
         this.text, {
         super.key,
         this.variant = FsmTextVariant.body,
         this.color,
         this.maxLines,
         this.overflow,
         this.textAlign,
       });

       final String text;
       final FsmTextVariant variant;
       final Color? color;
       final int? maxLines;
       final TextOverflow? overflow;
       final TextAlign? textAlign;

       @override
       Widget build(BuildContext context) {
         final theme = Theme.of(context);
         
         // вң… Automatically picks style from TextTheme
         final TextStyle? style = switch (variant) {
           FsmTextVariant.displayLarge => theme.textTheme.displayLarge,
           FsmTextVariant.displayMedium => theme.textTheme.displayMedium,
           FsmTextVariant.displaySmall => theme.textTheme.displaySmall,
           FsmTextVariant.headlineLarge => theme.textTheme.headlineLarge,
           FsmTextVariant.headlineMedium => theme.textTheme.headlineMedium,
           FsmTextVariant.headlineSmall => theme.textTheme.headlineSmall,
           FsmTextVariant.titleLarge => theme.textTheme.titleLarge,
           FsmTextVariant.titleMedium => theme.textTheme.titleMedium,
           FsmTextVariant.titleSmall => theme.textTheme.titleSmall,
           FsmTextVariant.bodyLarge => theme.textTheme.bodyLarge,
           FsmTextVariant.body => theme.textTheme.bodyMedium,  // Default
           FsmTextVariant.bodySmall => theme.textTheme.bodySmall,
           FsmTextVariant.labelLarge => theme.textTheme.labelLarge,
           FsmTextVariant.labelMedium => theme.textTheme.labelMedium,
           FsmTextVariant.labelSmall => theme.textTheme.labelSmall,
         };

         return Text(
           text,
           style: style?.copyWith(color: color),  // Allow color override
           maxLines: maxLines,
           overflow: overflow,
           textAlign: textAlign,
         );
       }
     }

     /// Text variants following Material 3 typography
     enum FsmTextVariant {
       displayLarge,    // 57sp - Hero titles
       displayMedium,   // 45sp - Large headers
       displaySmall,    // 36sp - Section headers
       
       headlineLarge,   // 32sp - Page titles
       headlineMedium,  // 28sp - Card headers
       headlineSmall,   // 24sp - Dialog titles
       
       titleLarge,      // 22sp - AppBar title
       titleMedium,     // 16sp - List item title
       titleSmall,      // 14sp - Compact titles
       
       bodyLarge,       // 16sp - Emphasized body
       body,            // 14sp - Default body (bodyMedium)
       bodySmall,       // 12sp - Captions
       
       labelLarge,      // 14sp - Button text
       labelMedium,     // 12sp - Chip labels
       labelSmall,      // 11sp - Small labels
     }

     /// Convenience constructors for common variants
     extension FsmTextConstructors on FsmText {
       /// Hero title (displayLarge)
       static FsmText hero(String text, {Key? key, Color? color}) =>
           FsmText(text, key: key, variant: FsmTextVariant.displayLarge, color: color);

       /// Page header (headlineLarge)
       static FsmText header(String text, {Key? key, Color? color}) =>
           FsmText(text, key: key, variant: FsmTextVariant.headlineLarge, color: color);

       /// Section title (headlineMedium)
       static FsmText title(String text, {Key? key, Color? color}) =>
           FsmText(text, key: key, variant: FsmTextVariant.headlineMedium, color: color);

       /// Body text (bodyMedium) - Default
       static FsmText body(String text, {Key? key, Color? color, int? maxLines}) =>
           FsmText(text, key: key, variant: FsmTextVariant.body, color: color, maxLines: maxLines);

       /// Caption text (bodySmall)
       static FsmText caption(String text, {Key? key, Color? color}) =>
           FsmText(text, key: key, variant: FsmTextVariant.bodySmall, color: color);

       /// Label text (labelMedium)
       static FsmText label(String text, {Key? key, Color? color}) =>
           FsmText(text, key: key, variant: FsmTextVariant.labelMedium, color: color);
     }
